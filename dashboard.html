    <div id="total-label">phones recycled</div>

    <div id="total-number">{{ total }}</div>

  </div>


  <script>

    const DURATION = {{ config.display_duration_seconds }} * 1000;

    const panel = document.getElementById("message-panel");

    const totalEl = document.getElementById("total-number");

    let hideTimer = null;


    function showMessage(total) {

      // Update counter with a bounce animation

      totalEl.textContent = total;

      totalEl.classList.add("bump");

      setTimeout(() => totalEl.classList.remove("bump"), 350);


      // Show thank you message then fade out

      panel.classList.add("visible");

      clearTimeout(hideTimer);

      hideTimer = setTimeout(() => panel.classList.remove("visible"), DURATION);

    }


    // Connect to real-time event stream

    const evtSource = new EventSource("/stream");

    evtSource.onmessage = (e) => {

      const data = JSON.parse(e.data);

      if (data.type === "deposit") showMessage(data.total);

    };

  </script>
